import requests import pandas as pd import json import os from datetime import datetime import traceback import io # 環境変数からデバッグモードを取得 DEBUG_MODE = os.getenv('DEBUG_MODE', 'false').lower() == 'true' DATA_FILE = 'data/shutai_data.json' TARGET_URL = 'https://nikkei225jp.com/data/shutai.php' def log(msg): print(f"[Scraper] {msg}", flush=True) def parse_value(val): """Pandasの値を数値に変換""" if pd.isna(val) or val == '-': return 0 s_val = str(val).strip() # 文字列のクリーニング s_val = s_val.replace(',', '').replace('%', '') # 記号処理（▼はマイナス、▲/+はプラス） if '▼' in s_val: s_val = '-' + s_val.replace('▼', '') elif '▲' in s_val: s_val = s_val.replace('▲', '') elif '+' in s_val: s_val = s_val.replace('+', '') try: if '.' in s_val: return float(s_val) return int(s_val) except ValueError: return 0 def scrape_shutai_data(): log(f"Starting scraping process using Pandas. Target: {TARGET_URL}") log(f"Debug Mode: {DEBUG_MODE}") headers = { 'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36', 'Accept-Language': 'ja,en-US;q=0.9,en;q=0.8' } try: # 1. HTMLの取得 response = requests.get(TARGET_URL, headers=headers, timeout=20) response.encoding = response.apparent_encoding if response.status_code != 200: log(f"Error: HTTP {response.status_code}") return html_content = response.text log(f"HTML Content Length: {len(html_content)}") # 2. Pandasでテーブル抽出 (lxml使用) # datatbl というIDを持つテーブルを探す try: dfs = pd.read_html(io.StringIO(html_content), attrs={'id': 'datatbl'}, flavor='lxml') except Exception as e: log(f"Pandas read_html failed: {e}") log("Trying bs4 flavor...") dfs = pd.read_html(io.StringIO(html_content), attrs={'id': 'datatbl'}, flavor='bs4') if not dfs: log("CRITICAL: No tables found matching id='datatbl'") return df = dfs[0] log(f"Table extracted. Shape: {df.shape}") # 3. データ処理 new_data = [] # 行ごとの処理 # dfの構造: # 0:日付, 1:日経平均, 2:変化, 3:海外, 4:証券自己 # 5:個人計, 6:現金, 7:信用, 8:投資信託, 9:事業法人, 10:その他, 11:信託, 12:生保, 13:都銀 for idx, row in df.iterrows(): # ヘッダー行や月計行の除外チェック # 日付カラムの値を確認 date_val = str(row.iloc[0]) # 日付フォーマットチェック (YYYY/MM/DD) try: # 2026/01/30 のような形式か、timeタグ内のテキスト # 余計な文字を除去 date_clean = date_val.split(' ')[0] # "2026/01/30" だけ取る date_obj = datetime.strptime(date_clean, '%Y/%m/%d') formatted_date = date_obj.strftime('%Y-%m-%d') except ValueError: # 日付変換できない行（ヘッダーや「月計」など）はスキップ continue try: # データのマッピング (ilocで位置指定) record = { "date": formatted_date, "nikkei_avg": parse_value(row.iloc[1]), "foreign": parse_value(row.iloc[3]), "securities_self": parse_value(row.iloc[4]), "individual_total": parse_value(row.iloc[5]), "individual_cash": parse_value(row.iloc[6]), "individual_credit": parse_value(row.iloc[7]), "investment_trust": parse_value(row.iloc[8]), "business_corp": parse_value(row.iloc[9]), "trust_banks": parse_value(row.iloc[11]) } new_data.append(record) except Exception as e: log(f"Error parsing row {idx}: {e}") continue log(f"Extracted {len(new_data)} valid records.") if len(new_data) == 0: log("WARNING: Valid data count is 0. Aborting save.") return # 4. 保存ロジック final_list = [] if DEBUG_MODE: log("DEBUG MODE: Overwriting ALL data.") final_list = sorted(new_data, key=lambda x: x['date']) else: log("NORMAL MODE: Merging data.") existing_data = [] if os.path.exists(DATA_FILE): try: with open(DATA_FILE, 'r', encoding='utf-8') as f: existing_data = json.load(f) except: pass data_map = {item['date']: item for item in existing_data} for item in new_data: data_map[item['date']] = item final_list = sorted(data_map.values(), key=lambda x: x['date']) # ディレクトリ作成と保存 os.makedirs(os.path.dirname(DATA_FILE), exist_ok=True) with open(DATA_FILE, 'w', encoding='utf-8') as f: json.dump(final_list, f, indent=4, ensure_ascii=False) log(f"Successfully saved to {DATA_FILE}") except Exception as e: log(f"FATAL ERROR: {e}") traceback.print_exc() if __name__ == "__main__": scrape_shutai_data()
